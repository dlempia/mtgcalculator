    $(window).on('load',function () {
        lazyLoadProductImages();
        loadFilterToggleBtns();
        updateSearchValue();
        initializeBackToTopIcon();

        var errorImagePath = '';
        var imageSources = $("img[id^='img_']");
        var i = 0;
        for (i = 0; i < imageSources.length; i++) {
            var image = $(imageSources[i]);
            var imageSource = $(image).attr('src');
            if (imageSource == errorImagePath) {
                var id = $(image).attr('id').replace('img_', '');
                $('#viewImage_' + id).attr('href', errorImagePath);
            }
        }

        window.onpopstate = popState;
    });

    function pushState() {
        var searchUrl = $("#SearchUrl").val();
        window.history.pushState({}, "Search Results - TCGplayer.com: Online Gaming Store for Cards, Miniatures, Singles, Packs & Booster Boxes", searchUrl);
        if(pageTracker)
            pageTracker._trackPageview(searchUrl);
    }

    function popState(event) {

        showLoadingMasks();

        $.ajax({
            method: "GET",
            url: document.location.href,
            cache: false,
            success: function (response) {
                processResult(response);
                lazyLoadProductImages();
                loadFilterToggleBtns();
            },
            error: function () {
                showError();
            },
            complete: function () {
                hideLoadingMasks();
            }
        });
    }

    function changeFilter(filterName, filterValue, productId, gameName) {

        showLoadingMasks();
        trackEvent("PLP Filters", filterName, filterValue);

        $.ajax({
            method: "GET",
            url: "https://shop.tcgplayer.com/productcatalog/product/ChangeShowFilter",
            data: "filterName=" + filterName + "&filterValue=" + filterValue + "&gameName=" + gameName,
            cache: false,
            success: function (response) {
                processResult(response);
                lazyLoadProductImages();
                loadFilterToggleBtns();
                pushState();
                _sift.push(['_trackPageview']);
            },
            error: function () {
                showError();
            },
            complete: function () {
                hideLoadingMasks();
            }
        });
    }

    function clearFilter(filterName, productId, gameName)
    {
        showLoadingMasks();

        $.ajax({
            method: "GET",
            url: "https://shop.tcgplayer.com/productcatalog/product/ClearShowFilter",
            data: "filterSection=" + filterName + "&gameName=" + gameName,
            cache: false,
            success: function (response) {
                processResult(response);
                lazyLoadProductImages();
                loadFilterToggleBtns();
                pushState();
            },
            error: function () {
                showError();
            },
            complete: function () {
                hideLoadingMasks();
            }
        });
    }

    function clearAllFilters(productId, gameName)
    {
        showLoadingMasks();

        $.ajax({
            method: "GET",
            url: "https://shop.tcgplayer.com/productcatalog/product/ClearAllShowFilters",
            data: "gameName=" + gameName,
        cache: false,
        success: function (response) {
            processResult(response);
            lazyLoadProductImages();
            loadFilterToggleBtns();
            pushState();
        },
        error: function () {
            showError();
        },
        complete: function () {
            hideLoadingMasks();
        }
    });
    }

    function changeSort(productSort, priceSort, gameName) {

        showLoadingMasks();

        $.ajax({
            method: "POST",
            url: "https://shop.tcgplayer.com/productcatalog/product/ChangeSort",
            data: "productSort=" + productSort + "&priceSort=" + priceSort + "&gameName=" + gameName,
            cache: false,
            success: function (response) {
                processResult(response);
                lazyLoadProductImages();
                loadFilterToggleBtns();
                pushState();
            },
            error: function () {
                showError();
            },
            complete: function () {
                hideLoadingMasks();
            }
        });
    }

    function changeProductResultDisplay(orientation, currentPage) {
        showLoadingMasks();

        var changeOrientationUrl = "/productcatalog/product/ChangeProductResultDisplay?orientation=" + orientation;
        if (pageTracker)
            pageTracker._trackPageview(changeOrientationUrl);

        $.ajax({
            method: "GET",
            url: changeOrientationUrl,
            cache: false,
            success: function (response) {
                var nextUrl = document.location.href.replace('&PageNumber=' + currentPage, '').replace('?PageNumber=' + currentPage, '');
                $.ajax({
                    method: "GET",
                    url: nextUrl,
                    cache: false,
                    success: function (response) {
                        processResult(response);
                        lazyLoadProductImages();
                        loadFilterToggleBtns();
                        pushState();
                    },
                    error: function () {
                        showError();
                    },
                    complete: function () {
                        hideLoadingMasks();
                    }
                });
            }
        });
    }

    function showLoadingMasks() {
        $('.search-results-container').addClass('loading-mask');
        $('.search-error-message').addClass('loading-mask');
        $('.product-filters').addClass('loading-mask');
    }

    function hideLoadingMasks() {
        $('.search-results-container').removeClass('loading-mask');
        $('.search-error-message').removeClass('loading-mask');
        $('.product-filters').removeClass('loading-mask');
    }

    function showError() {
        $('.search-results-container').hide();
        $('.search-error-message').show();
    }

    function hideError() {
        $('.search-error-message').hide();
        $('.search-results-container').show();
    }

    function processResult(response) {
        hideError();
        $(".product-filters").html(response.FilterView);
        $(".search-results-container").html(response.SearchResultsView);
        $(".search-results-header-container").html(response.SearchResultsHeaderView);
    }

    function lazyLoadProductImages() {
        $("img.lazy").lazyload({ threshold: 300 });
    }

    function loadFilterToggleBtns() {
        $('[data-toggle="show-filters"]').click(function () {
            $('[data-toggle="show-filters"]').toggleClass('active');
            $('.search-results__wrapper').toggleClass('show-filters');
            $('body').addClass('filters-open');
        });

        $('[data-toggle="hide-filters"]').click(function () {
            $('[data-toggle="show-filters"]').removeClass('active');
            $('.search-results__wrapper').removeClass('show-filters');
            $('body').removeClass('filters-open');
        });
    }

    function updateSearchValue() {

        $('.site-search__input').attr('value', searchTerm).attr('placeholder', '');
    }

        $("html, body").animate({ scrollTop: 0 }, "slow");
    }

        });
